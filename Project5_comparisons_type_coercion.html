//main.js // 1. Using typeof to display the data type of a variable document.write(typeof "Hello World"); document.write("
"); // 2. Type Coercion: combining a string and a number // This will result in "105" as JavaScript treats the 5 as text document.write("10" + 5); document.write("
"); // 3. Double Equal (==) Comparison // Returns true because the values are the same document.write(10 == 10); document.write("
"); // 4. Triple Equal (===) Comparison // Returns false because the data types are different (string vs number) document.write("10" === 10); document.write("
"); // 5. Greater Than (>) and Less Than (<) Operators document.write(5 > 2); // true document.write("
"); document.write(5 < 2); // false document.write("
"); // 6. Logical AND (&&) Operator // Returns true because both statements are true document.write(10 > 5 && 11 > 6); document.write("
"); // 7. Logical OR (||) Operator // Returns true because at least one statement is true document.write(5 > 10 || 10 > 4); document.write("
"); // 8. Logical NOT (!) Operator // Returns true because 5 is NOT greater than 10 document.write(!(5 > 10));
